 CREATE TABLE PHONG
 (
	MAPHONG CHAR(10) PRIMARY KEY,

 )
 GO

 CREATE TABLE NHANVIEN
 (
	MANV CHAR(10) PRIMARY KEY,
	TENNV VARCHAR(20),
	NGAYSINH DATE,
	VITRI VARCHAR(20)
 )

 )
 GO
